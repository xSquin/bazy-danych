SELECT PRODUCT_NAME, (AVG(ORDER_POSITION.SELL_PRICE) - PURCHASE_PRICE) AS "ROZNICA" FROM PRODUCT JOIN ORDER_POSITION ON PRODUCT.PRODUCT_ID = ORDER_POSITION.PRODUCT_ID GROUP BY PRODUCT.PRODUCT_ID ORDER BY ROZNICA DESC;

ZAD 2
1.
SELECT DISTINCT DEPARTMENT_NAME FROM DEPARTMENT JOIN EMPLOYEE ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID JOIN `ORDER` ON `ORDER`.EMPLOYEE_ID = EMPLOYEE.EMPLOYEE_NAME;


2.
SELECT DEPARTMENT_NAME, SUM(SELL_PRICE) FROM DEPARTMENT JOIN EMPLOYEE ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID JOIN `ORDER` ON EMPLOYEE.EMPLOYEE_NAME = `ORDER`.EMPLOYEE_ID JOIN ORDER_POSITION ON ORDER_POSITION.ORDER_ID = `ORDER`.ORDER_ID GROUP BY DEPARTMENT_NAME;

3.
SELECT DEPARTMENT_NAME, SUM(SELL_PRICE), STATUS_NAME FROM DEPARTMENT JOIN EMPLOYEE ON EMPLOYEE.DEPARTMENT_ID = DEPARTMENT.DEPARTMENT_ID JOIN `ORDER` ON EMPLOYEE.EMPLOYEE_NAME = `ORDER`.EMPLOYEE_ID JOIN ORDER_POSITION ON ORDER_POSITION.ORDER_ID = `ORDER`.ORDER_ID JOIN ORDER_STATUS ON ORDER_STATUS.STATUS_ID = `ORDER`.ORDER_STATUS GROUP BY DEPARTMENT_NAME, STATUS_NAME;
