--- zad 1 ---
1.
SELECT FULL_NAME, CLIENT_ADDRESS.STREET, LENGTH(CLIENT_ADDRESS.STREET), SUBSTR(CLIENT_ADDRESS.STREET, 1, 5) FROM CLIENT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID;
2.
SELECT SHORT_NAME, ADDRESS_TYPE_NAME FROM CLIENT LEFT JOIN CLIENT_ADDRESS ON CLIENT.CLIENT_ID = CLIENT_ADDRESS.CLIENT_ID LEFT JOIN ADDRESS_TYPE ON ADDRESS_TYPE.ADDRESS_TYPE_ID = CLIENT_ADDRESS.ADDRESS_TYPE;
3.
SELECT PRODUCT_NAME, CATEGORY_NAME, SUBSTR(PRODUCT_NAME, 1, 3) FROM PRODUCT JOIN CATEGORY ON PRODUCT.CATEGORY_ID = CATEGORY.CATEGORY_ID;
4.
SELECT PRODUCT_NAME, DESCRIPTION FROM PRODUCT JOIN ORDER_POSITION ON ORDER_POSITION.PRODUCT_ID = PRODUCT.PRODUCT_ID GROUP BY PRODUCT_NAME,DESCRIPTION;
--- zad 2 ---
1.
SELECT FULL_NAME, ORDER_NUMBER, DATEDIFF(CURDATE(), ORDER_DATE) FROM CLIENT JOIN `ORDER` ON `ORDER`.CLIENT_ID = CLIENT.CLIENT_ID;
2.
SELECT ORDER_ID, ORDER_DATE, DATE_ADD(ORDER_DATE, INTERVAL 14 DAY) AS ZAPLATA FROM `ORDER`  WHERE ORDER_STATUS NOT IN (3,7,10);
3.
SELECT FIRST_NAME, LAST_NAME, DATE_OF_EMPLOYMENT, DAYNAME(DATE_OF_EMPLOYMENT), DEPARTMENT_NAME FROM EMPLOYEE LEFT JOIN DEPARTMENT ON DEPARTMENT.DEPARTMENT_ID = EMPLOYEE.DEPARTMENT_ID;
4.
SELECT FULL_NAME, DATEDIFF(MAX(ORDER_DATE), MIN(ORDER_DATE)) FROM CLIENT
LEFT JOIN `ORDER` ON `ORDER`.CLIENT_ID = CLIENT.CLIENT_ID GROUP BY FULL_NAME;

1.
SELECT SHORT_NAME, GROUP_CONCAT(STREET SEPARATOR ', ') FROM CLIENT JOIN CLIENT_ADDRESS ON CLIENT_ADDRESS.CLIENT_ID = CLIENT.CLIENT_ID GROUP BY SHORT_NAME;
2.
SELECT CITY, GROUP_CONCAT(DISTINCT SUBSTR(POSTAL_CODE, 1, 2) SEPARATOR ',
 ') FROM CLIENT_ADDRESS GROUP BY CITY;
3.
SELECT CONCAT(FIRST_NAME,' ',LAST_NAME), DATEDIFF(CURDATE(), DATE_OF_EMPL
OYMENT), DEPARTMENT_NAME FROM EMPLOYEE JOIN DEPARTMENT ON DEPARTMENT.DEPARTMENT_ID = EMPLOYEE.DEPARTMENT_ID;
4.
SELECT CONCAT(FULL_NAME,'-',SHORT_NAME), GROUP_CONCAT(ORDER_NUMBER SEPARATOR ',') FROM CLIENT JOIN `ORDER` ON `ORDER`.CLIENT_ID = CLIENT.CLIENT_ID GROUP BY CONCAT(FULL_NAME,'-',SHORT_NAME);
5.
SELECT CONCAT(SUBSTR(FIRST_NAME, 1, 3), UPPER(RIGHT(LAST_NAME, 4)), LENGTH(FIRST_NAME)) AS HASLA FROM EMPLOYEE;
